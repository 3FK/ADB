CREATE TABLE ROVER_COMPUTER_Element(
Rover_Computer_Id VARCHAR(12),
Computer_OS varchar(50),
Computer_EEPROM varchar(10),
Computer_DRAM varchar(10),
Computer_Flash_Memory varchar(10),
Rover_ID varchar(10),
FOREIGN KEY (Rover_ID) REFERENCES ROVER(ROVER_ID),
CONSTRAINT PK_Rover_Size PRIMARY KEY (Rover_ID,Rover_Computer_Id)
)
/
CREATE SEQUENCE Rover_Computer_seq
START WITH 1
INCREMENT BY 1
NOCACHE
NOCYCLE;
/

CREATE OR REPLACE TRIGGER Rover_Computer_Id_Trigger
BEFORE INSERT ON ROVER_WHEELS
FOR EACH ROW
BEGIN
  SELECT 'RCID'|| to_char(Rover_Computer_seq.NEXTVAL,'00099')
  INTO   :new.Rover_Computer_Id
  FROM   dual;
END;
/
