CREATE TABLE THERMAL_SYSTEM(
Thermal_System_Id VARCHAR(12),

Rover_ID varchar(10),
FOREIGN KEY (Rover_ID) REFERENCES ROVER(ROVER_ID),
CONSTRAINT PK_Rover_Size PRIMARY KEY (Rover_ID,Rover_Size_Id)
)
/
CREATE SEQUENCE Rover_Size_seq
START WITH 1
INCREMENT BY 5
NOCACHE
NOCYCLE;
/

CREATE OR REPLACE TRIGGER Rover_Size_Id_Trigger
BEFORE INSERT ON ROVER_SIZE
FOR EACH ROW
BEGIN
  SELECT 'RSID'|| to_char(Rover_Size_seq.NEXTVAL,'00099')
  INTO   :new.Rover_Size_Id
  FROM   dual;
END;
/
